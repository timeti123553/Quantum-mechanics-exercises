> 求波函数 $\displaystyle\psi(r,\theta,\varphi)=c\mathrm{e}^{-\frac{r^{2}}{a^{2}}}\sin\theta\sin\varphi$ 的归一化系数 $c$，其中 $(r,\theta,\varphi)$ 为位矢 $x$ 的球坐标，并在 $\psi(r,\theta,\varphi)$ 状态下，求 $\displaystyle\frac{e^{2}}{r^{2}}$ 的平均值。

**解：**

波函数必须满足单值、连续和有限，即

$$
\begin{equation}\int\psi^*\psi\mathrm{d}^3x=1\end{equation}
$$

利用公式（1）求归一化系数

$$
\begin{equation}\begin{aligned}
\int\psi^*\psi\mathrm{d}^3x & =c^{2}\int_{r=0}^{\infty}\int_{\theta=0}^{\pi}\int_{\varphi=0}^{2\pi}\mathrm{e}^{-\frac{2r^{2}}{a^{2}}}\sin^{2}\theta\sin^{2}\varphi r^{2}\mathrm{d}r\sin\theta\mathrm{d}\theta\mathrm{d}\varphi \\
 & =c^{2}\int_{r=0}^{\infty}\mathrm{e}^{-\frac{2r^{2}}{a^{2}}}r^{2}\mathrm{d}r\int_{\theta=0}^{\pi}\sin^{3}\theta\mathrm{d}\theta\int_{\varphi=0}^{2\pi}\sin^{2}\varphi\mathrm{d}\varphi \\
 & =-c^{2}\int_{r=0}^{\infty}\mathrm{e}^{-\frac{2r^{2}}{a^{2}}}\frac{r}{2\left(-\frac{2}{a^{2}}\right)}\mathrm{d}\left(-\frac{2r^{2}}{a^{2}}\right)\int_{\theta=0}^{\pi}(1-\cos^{2}\theta)\mathrm{d}\cos\theta\int_{\varphi=0}^{2\pi}\frac{1-\cos2\varphi}{2}\mathrm{d}\varphi \\
 & =\frac{-c^{2}\pi}{-\frac{4}{a^{2}}}\left[r\mathrm{e}^{-\frac{2r^{2}}{a^{2}}}|_{0}^{\infty}-\int_{0}^{\infty}\mathrm{e}^{-\frac{2r^{2}}{a^{2}}}\mathrm{d}r\right]\left[\cos\theta|_{0}^{\pi}-\frac{1}{3}\cos^{3}\theta|_{0}^{\pi}\right]\\
 &=\frac{c^{2}a^{2}\pi}{4\sqrt{\frac{2}{a^{2}}}}\left(\frac{4}{3}\frac{\sqrt{\pi}}{2}\right)\\
 &=1
\end{aligned}\end{equation}
$$

故

$$
\begin{equation}
    \displaystyle\frac{c^2a^3\pi^{\frac{3}{2}}}{6\sqrt{2}}=1
\end{equation}
$$

$$
\begin{equation}
    \displaystyle c=\frac{6^{\frac{1}{2}}\times2^{\frac{1}{4}}}{a^{\frac{3}{2}}\pi^{\frac{3}{4}}}
\end{equation}
$$

$$
\begin{equation}\psi(r,\theta,\varphi)=\frac{6^{\frac{1}{2}}\times2^{\frac{1}{4}}}{a^{\frac{3}{2}}\pi^{\frac{3}{4}}}\mathrm{e}^{-\frac{r^{2}}{a^{2}}}\sin\theta\sin\varphi\end{equation}
$$

再求 $\displaystyle\frac{e^{2}}{r^{2}}$ 的平均值。

$$
\begin{equation}\mid\psi(r,\theta,\varphi)\mid^2=\frac{6\sqrt{2}}{a^3\pi^{\frac{3}{2}}}\mathrm{e}^{-\frac{2r^2}{a^2}}\sin^2\theta\sin^2\varphi\end{equation}
$$

则

$$
\begin{equation}\begin{aligned}
\left\langle\frac{e^2}{r^2}\right\rangle & =\frac{6\sqrt{2}}{a^{3}\pi^{\frac{3}{2}}}\int_{r=0}^{\infty}\int_{\theta=0}^{\pi}\int_{\varphi=0}^{2\pi}\frac{e^{2}}{r^{2}}\mathrm{e}^{-\frac{2r^{2}}{a^{2}}}\sin^{2}\theta\sin^{2}\varphi r^{2}\sin\theta\mathrm{d}r\mathrm{d}\theta\mathrm{d}\varphi \\
 & =\frac{6\sqrt{2}e^{2}}{a^{3}\pi^{\frac{3}{2}}}\int_{r=0}^{\infty}\mathrm{e}^{-\frac{2r^{2}}{a^{2}}}\mathrm{d}r\int_{\theta=0}^{\pi}\sin^{3}\theta\mathrm{d}\theta\int_{\varphi=0}^{2\pi}\sin^{2}\varphi\mathrm{d}\varphi \\
 & =-\frac{6\sqrt{2}e^{2}}{a^{3}\pi^{\frac{3}{2}}}\frac{1}{\sqrt{\frac{2}{a^{2}}}}\frac{\sqrt{\pi}}{2}\int_{\theta=0}^{\pi}(1-\cos^{2}\theta)\mathrm{d}\cos\theta\int_{\varphi=0}^{2\pi}\frac{1-\cos2\varphi}{2}\mathrm{d}\varphi \\
 & =-\frac{6\sqrt{2}e^{2}}{a^{3}\pi^{\frac{3}{2}}}\frac{1}{\sqrt{\frac{2}{a^{2}}}}\frac{\sqrt{\pi}}{2}\int_{\theta=0}^{\pi}(1-\cos^{2}\theta)\mathrm{d}\cos\theta\int_{\varphi=0}^{2\pi}\frac{1-\cos2\varphi}{2}\mathrm{d}\varphi \\
 & =-\frac{3e^{2}}{a^{2}}\left[\cos|_{0}^{\pi}-\frac{1}{3}\cos^{3}\theta|_{0}^{\pi}\right]\\
 &=\frac{4e^{2}}{a^{2}}
\end{aligned}\end{equation}
$$

所以，归一化系数 $c$ 为 $\displaystyle\frac{6^\frac{1}{2}\times2^\frac{1}{4}}{a^\frac{3}{2}\pi^\frac{3}{4}}$，$\displaystyle\frac{e^2}{r^2}$ 的平均值为 $\displaystyle\frac{4e^2}{a^2}$。
