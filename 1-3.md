> 如果系统处于在态 $\psi(x)= \begin{cases} Ax\mathrm{e}^{-\lambda x}, & x\geqslant0 \\ 0, & x<0 & \end{cases}$，求归一化常数 $A$、平均位置 $\left\langle x\right\rangle$、动量分布的概率幅 $\mathit{\phi}(p)$ 以及动量平均值 $\left\langle p\right\rangle$。

**解：**

利用归一化条件

$$
\begin{equation}\int\psi^*\psi\mathrm{d}^3x=1\end{equation}
$$

我们可以求出归一化常数。

其他变量的求解公式为：

$$
\begin{equation}\left\langle x\right\rangle=\int_{-\infty}^{\infty}x\psi^{*}\psi\mathrm{d}x\end{equation}
$$

$$
\begin{equation}\mathit{\phi}(p)=\int_{-\infty}^{\infty}\psi^{*}{}_{p}(x)\psi(x)\mathrm{d}x\end{equation}
$$

$$
\begin{equation}\left\langle p\right\rangle=\int_{-\infty}^{\infty}\Phi^{*}(p)p\Phi(p)\mathrm{d}p\end{equation}
$$

由公式（1），我们可以得到

$$
\begin{equation}\int_0^\infty A^2x^2\mathrm{e}^{-2\lambda x}\mathrm{d}x=1\end{equation}
$$

$$
\begin{equation}\frac{A^2}{-2\lambda}\int_0^\infty x^2\mathrm{d}\mathrm{e}^{-2\lambda x}=1\end{equation}
$$

$$
\begin{equation}
\frac{A^2}{-2\lambda}\left[x^2\mathrm{e}^{-2\lambda x}|_0^\infty-\int_0^\infty\mathrm{e}^{-2\lambda x}2x\mathrm{d}x\right]=1
\end{equation}
$$

$$
\begin{equation}\frac{A^2}{2\lambda(-2\lambda)}\int_0^\infty2x\mathrm{de}^{-2\lambda x}=1\end{equation}
$$

$$
\begin{equation}\frac{A^2}{\lambda(-2\lambda)}\left[x\mathrm{e}^{-2\lambda x}|_0^\infty-\int_0^\infty\mathrm{e}^{-2\lambda x}\mathrm{d}x\right]=1\end{equation}
$$

$$
\begin{equation}\frac{A^2}{\lambda(2\lambda)(-2\lambda)}\mathrm{e}^{-2\lambda x}|_0^\infty=1\end{equation}
$$

最后，我们得到归一化常数 $A$ 为

$$
\begin{equation}A=2\sqrt{\lambda^3}\end{equation}
$$

由公式（2），我们可以得到

$$
\begin{equation}\begin{aligned}
\left\langle x\right\rangle & =\int_{0}^{\infty}xA^{2}x^{2}\mathrm{e}^{-2\lambda x}\mathrm{d}x\\
&=4\lambda^{3}\int_{0}^{\infty}x^{3}\mathrm{e}^{-2\lambda x}\mathrm{d}x\\
&=-\frac{4\lambda^{3}}{2\lambda}\int_{0}^{\infty}x^{3}\mathrm{d}\mathrm{e}^{-2\lambda x} \\
 & =-\frac{4\lambda^{3}}{2\lambda}\left[x^{3}\mathrm{e}^{-2\lambda x}|_{0}^{\infty}-\int_{0}^{\infty}\mathrm{e}^{-2\lambda x}3x^{2}\mathrm{d}x\right] \\
 & =-\frac{6\lambda^{2}}{2\lambda}\int_{0}^{\infty}x^{2}\mathrm{d}\mathrm{e}^{-2\lambda x}\\
 &=-\frac{6\lambda^{2}}{2\lambda}\left[x^{2}\mathrm{e}^{-2\lambda x}|_{0}^{\infty}-\int_{0}^{\infty}\mathrm{e}^{-2\lambda x}2x\mathrm{d}x\right]\\
 &=-\frac{6\lambda}{2\lambda}\left[\int_{0}^{\infty}x\mathrm{d}\mathrm{e}^{-2\lambda x}\right] \\
 & =-\frac{6\lambda}{2\lambda}\left[x\mathrm{e}^{-2\lambda x}|_{0}^{\infty}-\int_{0}^{\infty}\mathrm{e}^{-2\lambda x}\mathrm{d}x\right]\\
 &=-\frac{3}{2\lambda}\left[\int_{0}^{\infty}\mathrm{d}\mathrm{e}^{-2\lambda x}\right]\\
 &=-\frac{3}{2\lambda}\mathrm{e}^{-2\lambda x}|_{0}^{\infty}\\
 &=\frac{3}{2\lambda}
\end{aligned}\end{equation}
$$

根据公式（3），我们求出动量分布的概率幅 $\phi(p)$。

其中 

$$
\begin{equation}\psi_{p}(x)=\frac{1}{\sqrt{2\pi\hbar}}\mathrm{e}^{\frac{\mathrm{i}}{\hbar}px}\end{equation}
$$

我们可以得到

$$
\begin{equation}\begin{aligned}
\phi(p) & =\int_{-\infty}^{\infty}\psi_{p}(x)\psi(x)\mathrm{d}x\\
&=\int_{0}^{\infty}\frac{1}{\sqrt{2\pi\hbar}}\mathrm{e}^{-\frac{\mathrm{i}}{\hbar}px}2\sqrt{\lambda^{3}}x\mathrm{e}^{-\lambda x}\mathrm{d}x \\
 & =\frac{\sqrt{2\lambda^3}}{\sqrt{\pi\hbar}}\frac{1}{-\left(\frac{\mathrm{i}}{\hbar}p+\lambda\right)}\int_0^\infty x\mathrm{de}^{-x\left(\frac{\mathrm{i}}{\hbar}p+\lambda\right)} \\
 & =\frac{\sqrt{2\lambda^{3}}}{\sqrt{\pi\hbar}}\frac{1}{-\left(\frac{\mathrm{i}}{\hbar}p+\lambda\right)}\left[x\mathrm{e}^{-x\left(\frac{\mathrm{i}}{\hbar}p+\lambda\right)}\bigg|_{0}^{\infty}-\int_{0}^{\infty}\mathrm{e}^{-x\left(\frac{\mathrm{i}}{\hbar}p+\lambda\right)}\mathrm{d}x\right] \\
 & =\frac{\sqrt{2\lambda^{3}}}{\sqrt{\pi\hbar}}\frac{1}{-\left(\frac{\mathrm{i}}{\hbar}p+\lambda\right)^{2}}\mathrm{e}^{-x\left(\frac{\mathrm{i}}{\hbar}p+\lambda\right)} \bigg|_{0}^{\infty}\\
 &=\frac{\sqrt{2\lambda^{3}}}{\sqrt{\pi\hbar}}\frac{1}{\left(\frac{\mathrm{i}}{\hbar}p+\lambda\right)^{2}}
\end{aligned}\end{equation}
$$

根据公式（4），我们可以求出动量平均值

$$
\begin{equation}
\begin{aligned}
\left\langle p\right\rangle
&=\int_{-\infty}^{\infty}\Phi^{*}(p)p\Phi(p)\mathrm{d}\\
&=\frac{2\lambda^{3}}{\pi\hbar}\int_{-\infty}^{\infty}\frac{p\mathrm{d}p}{\left(\frac{\mathrm{i}}{\hbar}p+\lambda\right)^{2}\left(-\frac{\mathrm{i}}{\hbar}p+\lambda\right)^{2}}\\
&=\frac{\lambda^{3}\hbar}{\pi}\int_{-\infty}^{\infty}\frac{\mathrm{d}\frac{p^{2}}{\hbar^{2}}}{\left[\left(\frac{p}{\hbar}\right)^{2}+\lambda^{2}\right]^{2}}\\
&=-\frac{\lambda^{3}\hbar}{\pi}\frac{1}{\left(\frac{p}{\hbar}\right)^{2}+\lambda^{2}}\bigg|_{p=-\infty}^{\infty}\\
&=0
\end{aligned}
\end{equation}
$$

**归一化常数：**

$$
\begin{equation}
    A=2\sqrt{\lambda^3}
\end{equation}
$$
 
**平均位置：**

$$
\begin{equation}
    \left\langle x\right\rangle=\frac{3}{2\lambda}
\end{equation}
$$

**动量分布的概率幅：**

$$
\begin{equation}
    \phi(p)=\frac{\sqrt{2\lambda^3}}{\sqrt{\pi\hbar}}\frac{1}{\left(\frac{\mathrm{i}}{\hbar}p+\lambda\right)^2}
\end{equation}
$$

**动量平均值：**

$$
\begin{equation}
    \left\langle p\right\rangle=0
\end{equation}
$$
